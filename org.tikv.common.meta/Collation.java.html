  public static int translate(String collation) {
    Integer code = collationMap.get(collation);
    if (code == null) {
      return DEF_COLLATION_CODE;
    }
    return code;
  }

  public static String translate(int code) {

    if (collation == null) {
      return "";
    }
    return collation;
